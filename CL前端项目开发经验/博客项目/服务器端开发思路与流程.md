## 5.对哪些接口要权限验证

```js
1. 不可能把所有的接口都暴露给别人，把那些可以操作数据库的接口全部需要登录验证，只有你登录过才可以操作那些接口

3. 做权限验证的方式有三，第一利用cookie，  第二利用session，  第三利用jwt
```



### 5.1 JWT做权限验证

```js
//1. 专门用来封装jwt的

const jwt = require('jsonwebtoken');

const secret = 'caolei';

module.exports = {
    /**
    * 利用jwt颁发
    **/
    sign: function(res, payload = {}, maxAge = 24 * 3600){
        const token = jwt.sign(payload, secret, {
            expiresIn: maxAge          //过期时间
        })
        
        res.cookie('token', token, {
            //token的一些额外的配置
            maxAge,
            httponly: true,
            path: '/'
        });
        
        //防止不是客户端
        res.header('authorization', token);
    },
    
    /**
    * 验证jwt
    **/
    verify: function(token){
        try{
            return jwt.verify(token, secret);
        }catch(err){
            console.log(err);
        }
    }
}

//2. 处理jwt的中间件

const regPath = require('path-to-expreg');   //把路径变成正则表达式的形式
const jwt = require(../jwt);
//免登录的白名单
const WhiteList = [
    
]

module.exports = (req, res, next) => {
    const isToken = withoutToken(req);
    
    if(isToken.length > 0){
        //不用登录
        next();
        return;
    }
    
    let token = req.cookies.token;
    if(token){
        token = req.headers['authorization'];
        if(!token){
            res.status(403).send('没有权限哦');
            return;
        }
        token = token.split(' ');
        token = token.length == 1? token[0] : token[1];
    }
    
    //验证token
    const result = jwt.verify(token);
    if(!result){
        res.status(403).send('没有权限哦');
        return;
    }
    
    next();
}

function withoutToken(req){
    return WhiteList.filter(item => {
        const reg = regPath(item.path);
        return item.method == req.method && reg.test(req.path) 
    })
}


//3. 使用这个中间件， 一定要使用  cookie-parser这个中间件对cookie进行解析
```
