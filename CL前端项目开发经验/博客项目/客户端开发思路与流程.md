#### 2.服务端支持跨域CORS(cross origin resource sharing)

1. ##### 自己写CORS跨域中间件

```js
const WhiteList = [];    //允许跨域的白名单

module.exports = (req, res, next) => {
    //是否是预检请求，请求方法是options, 如果是预检请求的话，需要在响应头里面添加这几个字段
    if(req.method == 'OPTIONS'){
        res.header('Access-Control-Allow-Methods', req.headers['access-control-request-method']);
        res.header('Access-Control-Allow-Headers', req.headers['access-control-request-headers']);
        res.header('Access-Control-Allow-Max-Age', 3600 * 24);
    }
    
    //简单请求， 请求方法只限于get, post, header, 简单的请求头,  会在请求头带有origin字段
    if('origin' in req.headers && WhiteList.includes(req.headers['origin'])){
        req.header('Access-Control-Allow-Origin', req.headers['origin'])
    }
}
```



##### 2.npm i cors

```js
const express = require('express');

const cors = require('cors');

const app = express();

const WhiteList = [];

const corsOptions = {
    origin: function(origin, callback){
        if(whiteList.indexOf(origin) !== -1){
            callback(null, true);
        }else{
            callback(new Error('Not Allow By CORS'));
        }
    }
}

app.use(cors(corsOptions));
```
